149inline auto sat_overlap(const std::vector< glm::vec2 >& a, const std::vector< glm::vec2 >& b) -> bool
150{
151 auto edge_normal = [](const glm::vec2& a, const glm::vec2& b) {
152 glm::vec2 edge{b - a};
153 return glm::normalize(glm::vec2{-edge.y, edge.x});
154 };
155
156 auto project = [](const std::vector< glm::vec2 >& points, glm::vec2 normal) {
157 float min{INFINITY}, max{-INFINITY};
158 for (const auto& p : points) {
159 min = std::min(min, glm::dot(p, normal));
160 max = std::max(max, glm::dot(p, normal));
161 }
162 return std::make_pair(min, max);
163 };
164
165 auto detect_overlap = [&](const auto& convex_hull) {
166 for (unsigned i = 0; i < convex_hull.size(); i++) {
167 const glm::vec2 axis{edge_normal(convex_hull[i], convex_hull[(i + 1) % convex_hull.size()])};
168 auto [a_min, a_max]{project(a, axis)};
169 auto [b_min, b_max]{project(b, axis)};
170 if (a_min > b_max || b_min > a_max)
171 return false;
172 }
173 return true;
174 };
175
176 return detect_overlap(a) && detect_overlap(b);
177}
