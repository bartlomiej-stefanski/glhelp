76template< VertexType Vertex >
77auto Obj< Vertex >::parse_from_file(std::istream& in_stream) -> Obj< Vertex >
78{
79 ParseState parse_state{};
80
81 std::string line;
82 while (std::getline(in_stream, line)) {
83 parse_state.current_line++;
84 if (line.empty()) {
85 continue;
86 }
87
88 std::stringstream ss(std::move(line));
89 parse_line(ss, parse_state);
90 }
91
92 Obj< Vertex > obj;
93 std::unordered_map< ParseState::VertexData, std::size_t > deduplicator;
94 for (const auto& vertex: parse_state.vertices) {
95 if (!deduplicator.contains(vertex)) {
96 obj.vertices.emplace_back(Obj< Vertex >::create_vertex(parse_state, vertex));
97 deduplicator[vertex] = obj.vertices.size() - 1;
98 }
99
100 obj.indices.emplace_back(deduplicator.at(vertex));
101 }
102
103 parse_state.log_warnings(std::cerr);
104
105 obj.obj_name = std::move(parse_state.obj_name);
106 return obj;
107}
